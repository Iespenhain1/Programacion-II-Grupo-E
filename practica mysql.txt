create database clinica;
use clinica;
create table pacientes(
	id_pacientes int auto_increment primary key,
    nombre VARCHAR(100),
    edad int,
    genero varchar(10),
    diagnostico varchar(150)
);

delete from pacientes where id_pacientes=1;
